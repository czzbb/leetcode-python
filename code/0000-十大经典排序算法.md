# 冒泡排序
每次冒泡最大的的放到后面
```python
def popsort(arr):
    for i in range(1, len(arr)):
        for j in range(0, len(arr)-i):
            if arr[j]>arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
    return arr
```

# 选择排序
每次选最小的放到前面
```python
def selectsort(arr):
    for i in range(len(arr)-1):
        minflag = i
        for j in (i+1, len(arr)):
            if arr[minflag] > arr[j]:
                minflag = j
        if minflag != i:
            arr[i], arr[minflag] = arr[minflag], arr[i]
    return arr
```

# 插入排序
```python
def insertsor(arr):
    for i in range(1, len(arr)):
        pre = i-1
        current = arr[i]
        while pre>=0 and current < arr[pre]:
            arr[pre+1] = arr[pre]
            pre -= 1
        arr[pre+1] = current
    return arr
```
