class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head: return head
        top = ListNode(-1)
        top.next = head
        cur = top
        while cur.next and cur.next.next:
            n1 = cur.next
            n2 = cur.next.next
            n1.next = n2.next
            n2.next = n1
            cur.next = n2
            cur = n1
        return top.next
